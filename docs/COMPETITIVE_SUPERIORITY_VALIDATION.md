function detectPlatform(meetingUrl) {

    // Existing patterns...
    skype: /skype\.com\/join/,
    dialpad: /dialpad\.com\/meeting/,

    jitsi: /meet\.jit\.si/,

  };

    if (pattern.test(meetingUrl)) {
      return platform;
    }
  }
  return 'generic';
